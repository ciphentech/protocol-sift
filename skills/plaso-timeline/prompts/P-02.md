   from dataclasses import dataclass, asdict
   from enum import IntEnum
   from typing import Optional
   import math
   import re

   class ConfidenceRank(IntEnum):
       """Confidence ranks for NTP source resolution (spec.md §4 Phase 2)."""
       ARTIFACT_LOG     = 1   # EID 35/37/259/260 recovered from artifact
       ANALYST_VERIFIED = 2   # --ntp-source provided and cross-checked vs logs
       ANALYST_STATED   = 3   # --ntp-source provided, no logs to verify
       CLOUD_DEFAULT    = 4   # cloud provider environment default (AWS/Azure/GCP/etc.)
       WINDOWS_DOMAIN   = 5   # On-prem Windows domain assumption
       DISTRO_DEFAULT   = 6   # Linux distro / unknown fallback

   @dataclass(frozen=True, slots=True)
   class NTPContext:
       ntp_source: str = ""
       ntp_offset_s: float = 0.0
       ntp_assumption: bool = True
       confidence_rank: int = 6   # int in [1, 6] or ConfidenceRank
       source_eid: Optional[int] = None   # EID that provided the source, if any

       def __post_init__(self) -> None:
           # Type check — declarative hints aren't enforced at runtime.
           # bool excluded because isinstance(True, int) is True in Python.
           if isinstance(self.ntp_offset_s, bool) or \
              not isinstance(self.ntp_offset_s, (int, float)):
               raise TypeError(
                   f"ntp_offset_s must be numeric, "
                   f"got {type(self.ntp_offset_s).__name__}"
               )
           # NaN/inf are not meaningful clock offsets. The plausibility bound
           # (±1000 s) lives in validate_context() in P-05 so the
           # self-correction loop can recover from implausible offsets rather
           # than crash at construction.
           if math.isnan(self.ntp_offset_s) or math.isinf(self.ntp_offset_s):
               raise ValueError(
                   f"ntp_offset_s must be finite, got {self.ntp_offset_s}"
               )
           # confidence_rank must be in the 1–6 range documented in spec §4
           if not (1 <= int(self.confidence_rank) <= 6):
               raise ValueError(
                   f"confidence_rank must be 1 through 6, "
                   f"got {self.confidence_rank}"
               )
           # source_eid must be a known Windows Time Service event ID
           if self.source_eid is not None and \
              self.source_eid not in (35, 37, 259, 260):
               raise ValueError(
                   f"source_eid must be one of 35/37/259/260 or None, "
                   f"got {self.source_eid}"
               )
           # Confirmed sources cannot be blank — ntp_assumption=False
           # represents a *known* source.
           if self.ntp_assumption is False and not self.ntp_source:
               raise ValueError(
                   "ntp_assumption=False requires a non-empty ntp_source"
               )

       def to_dict(self) -> dict:
           """Serialize for manifest JSON. confidence_rank coerced to plain int."""
           d = asdict(self)
           d["confidence_rank"] = int(self.confidence_rank)
           return d

   # --- EID extraction helpers (all return None on no match) ---

   EID_PATTERN = re.compile(r'\[(\d+)\s*/\s*0x')   # matches "[260 / 0x0104]" in short/desc

   def extract_eid_number(row: dict) -> Optional[int]:
       """Return the Windows Event ID from the short or desc column, or None."""
       for col in ('short', 'desc'):
           m = EID_PATTERN.search(row.get(col, ''))
           if m:
               return int(m.group(1))
       return None

   PHASE_OFFSET_PATTERN = re.compile(r'Phase Offset:\s*([+-]?\d+\.\d+)s')

   def extract_phase_offset_s(desc: str) -> Optional[float]:
       """Extract PhaseOffset in seconds from EID 260 desc string.
       Strips trailing 's', converts to float.
       Returns None if pattern not found."""
       m = PHASE_OFFSET_PATTERN.search(desc)
       return float(m.group(1)) if m else None

   NTP_SOURCE_PATTERN = re.compile(
       r"Strings:\s*\['([a-zA-Z0-9._-]+)\s*(?:\([^)]*\))?[;']"
   )

   def extract_ntp_source_hostname(desc: str) -> Optional[str]:
       """Extract the NTP server hostname from EID 37, 259 desc Strings field.
       Returns bare hostname only (no port info).
       Returns None if pattern not found."""
       m = NTP_SOURCE_PATTERN.search(desc)
       return m.group(1).strip() if m else None
