def resolve_ntp_source(df, cli_ntp_source: Optional[str] = None,
                       skip_ntp: bool = False,
                       interactive: bool = False) -> NTPContext:
    """
    Resolution priority (spec.md §4 Phase 2):
      1. Artifact NTP logs (EID 35, 37, 259, 260) — confidence_rank 1
      2. --ntp-source CLI flag — confidence_rank 2
      3. Interactive analyst prompt — confidence_rank 3
      4. Cloud provider default — confidence_rank 4
      5. On-prem Windows domain default — confidence_rank 5
      6. Linux distro / unknown default — confidence_rank 6

    If skip_ntp=True, returns a null NTPContext with ntp_source="" so the
    caller can detect and pass through without enrichment.

    df: pandas DataFrame with l2tcsv columns.
    """
    if skip_ntp:
        return NTPContext(ntp_source="", ntp_assumption=True, confidence_rank=6)

    # Priority 1: scan artifact for EID 260 (best — has offset too)
    eid260_rows = df[df.apply(lambda r: extract_eid_number(r) == 260, axis=1)]
    if not eid260_rows.empty:
        first = eid260_rows.iloc[0]
        source = extract_ntp_source_hostname(first.get('desc', ''))  # EID 260 has Source field
        offset = extract_phase_offset_s(first.get('desc', ''))
        if source and offset is not None:
            return NTPContext(ntp_source=source, ntp_offset_s=offset,
                              ntp_assumption=False, confidence_rank=1, source_eid=260)

    # Priority 1b: EID 259 / 37 (hostname only; offset carried from EID 260 scan above)
    for eid in (259, 37):
        eid_rows = df[df.apply(lambda r: extract_eid_number(r) == eid, axis=1)]
        if not eid_rows.empty:
            source = extract_ntp_source_hostname(eid_rows.iloc[0].get('desc', ''))
            if source:
                # Try to get offset from any EID 260 in the same df
                offset = 0.0
                if not eid260_rows.empty:
                    offset = extract_phase_offset_s(
                        eid260_rows.iloc[0].get('desc', '')) or 0.0
                return NTPContext(ntp_source=source, ntp_offset_s=offset,
                                  ntp_assumption=False, confidence_rank=1, source_eid=eid)

    # Priority 2: CLI flag
    if cli_ntp_source:
        return NTPContext(ntp_source=cli_ntp_source, ntp_offset_s=0.0,
                          ntp_assumption=False, confidence_rank=2)

    # Priority 3: interactive prompt (only when interactive=True)
    if interactive:
        answer = input("[ntp-enrichment] NTP source for this host? (Enter to skip): ").strip()
        if answer:
            return NTPContext(ntp_source=answer, ntp_offset_s=0.0,
                              ntp_assumption=False, confidence_rank=3)

    # Priority 4–6: environment defaults (assumption=True)
    # For this PoC, default to domain controller pattern observed in artifact
    return NTPContext(ntp_source="time.windows.com", ntp_offset_s=0.0,
                      ntp_assumption=True, confidence_rank=5)
