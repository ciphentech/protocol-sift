import csv
import hashlib
import re
import sys
from datetime import datetime, timedelta, timezone
from pathlib import Path
from typing import Optional
import pandas as pd

from ntp_resolver import NTPContext, extract_eid_number, extract_phase_offset_s

# Paths that must never be written to (evidence integrity)
_FORBIDDEN_OUTPUT_PREFIXES = ('/cases/', '/mnt/', '/media/')

NEW_COLUMNS = ['ntp_source', 'ntp_offset_s', 'nist_delta_s', 'nist_time', 'ntp_assumption']
PLASO_DATE_FMT = '%m/%d/%Y'
PLASO_TIME_FMT = '%H:%M:%S'


def _sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with open(path, 'rb') as f:
        for chunk in iter(lambda: f.read(65536), b''):
            h.update(chunk)
    return h.hexdigest()


def _parse_event_ts(row: dict) -> Optional[datetime]:
    """Parse date+time columns into an aware UTC datetime. Returns None on error."""
    try:
        dt_str = f"{row['date']} {row['time']}"
        naive = datetime.strptime(dt_str, f'{PLASO_DATE_FMT} {PLASO_TIME_FMT}')
        return naive.replace(tzinfo=timezone.utc)
    except (KeyError, ValueError):
        return None


def _compute_nist_time(event_ts: datetime, offset_s: float) -> datetime:
    """nist_time = event_ts - timedelta(seconds=offset_s).
    Positive offset means the local clock was ahead of NIST → subtract to get true time.
    """
    return event_ts - timedelta(seconds=offset_s)


def enrich(csv_path: str | Path,
           ctx: NTPContext,
           output_path: str | Path,
           chunksize: int = 10_000) -> dict:
    """
    Read csv_path (read-only), compute five NTP fields per row,
    write enriched CSV to output_path sorted on nist_time.

    Returns a summary dict:
      {'rows_processed': N, 'source_hash_before': H, 'source_hash_after': H,
       'integrity_ok': bool, 'output_path': str}

    Raises ValueError if output_path is under a forbidden evidence prefix.
    Raises RuntimeError if source file hash changes (spoliation check).
    """
    csv_path = Path(csv_path)
    output_path = Path(output_path)

    # Evidence integrity: reject forbidden output paths
    for prefix in _FORBIDDEN_OUTPUT_PREFIXES:
        if str(output_path).startswith(prefix):
            raise ValueError(
                f"Output path {output_path} is inside a protected evidence "
                f"directory ({prefix}). Write to ./exports/ instead."
            )

    # Hash source before processing
    hash_before = _sha256_file(csv_path)

    # If skip-ntp (null context), copy through with no new columns
    if not ctx.ntp_source:
        import shutil
        shutil.copy2(csv_path, output_path)
        hash_after = _sha256_file(csv_path)
        return {'rows_processed': 0, 'source_hash_before': hash_before,
                'source_hash_after': hash_after, 'integrity_ok': True,
                'output_path': str(output_path)}

    # Read full CSV (chunked for large files)
    chunks = []
    for chunk in pd.read_csv(csv_path, chunksize=chunksize, dtype=str,
                              keep_default_na=False):
        chunks.append(chunk)
    df = pd.concat(chunks, ignore_index=True) if chunks else pd.DataFrame()

    rows_out = []
    for _, row in df.iterrows():
        r = dict(row)

        # Per-event offset: use EID 260 value if this row IS an EID 260 event
        eid = extract_eid_number(r)
        if eid == 260:
            per_event_offset = extract_phase_offset_s(r.get('desc', ''))
            offset = per_event_offset if per_event_offset is not None else ctx.ntp_offset_s
        else:
            offset = ctx.ntp_offset_s

        event_ts = _parse_event_ts(r)
        if event_ts:
            nist_ts = _compute_nist_time(event_ts, offset)
            r['nist_time'] = nist_ts.strftime('%Y-%m-%d %H:%M:%S.%f')[:-3] + 'Z'
        else:
            r['nist_time'] = ''

        r['ntp_source']    = ctx.ntp_source
        r['ntp_offset_s']  = round(offset, 7)
        r['nist_delta_s']  = round(offset, 7)   # spec §2.2: nist_delta_s == ntp_offset_s
        r['ntp_assumption']= ctx.ntp_assumption
        rows_out.append(r)

    # Sort on nist_time
    out_df = pd.DataFrame(rows_out)
    if 'nist_time' in out_df.columns and not out_df.empty:
        out_df = out_df.sort_values('nist_time', kind='stable')

    out_df.to_csv(output_path, index=False)

    # Spoliation check: source hash must be unchanged
    hash_after = _sha256_file(csv_path)
    integrity_ok = (hash_before == hash_after)
    if not integrity_ok:
        raise RuntimeError(
            f"EVIDENCE INTEGRITY VIOLATION: {csv_path} was modified during "
            f"enrichment. Before={hash_before[:12]}… After={hash_after[:12]}…"
        )

    return {
        'rows_processed': len(out_df),
        'source_hash_before': hash_before,
        'source_hash_after': hash_after,
        'integrity_ok': integrity_ok,
        'output_path': str(output_path),
    }

test_nist_time_negative_offset_shifts_forward:
  ctx = NTPContext(ntp_source="dc01", ntp_offset_s=-0.0004007, ntp_assumption=False)
  # nist_time = event_ts - (-0.0004007) = event_ts + 0.0004007
  event_ts = datetime(2018, 8, 30, 2, 5, 15, tzinfo=timezone.utc)
  result = _compute_nist_time(event_ts, -0.0004007)
  assert result > event_ts

test_nist_time_positive_offset_shifts_backward:
  ctx = NTPContext(ntp_source="dc01", ntp_offset_s=0.0032941, ntp_assumption=False)
  event_ts = datetime(2018, 8, 30, 10, 20, 13, tzinfo=timezone.utc)
  result = _compute_nist_time(event_ts, 0.0032941)
  assert result < event_ts

test_nist_time_zero_offset_unchanged:
  event_ts = datetime(2023, 1, 25, 14, 52, 4, tzinfo=timezone.utc)
  result = _compute_nist_time(event_ts, 0.0)
  assert result == event_ts
