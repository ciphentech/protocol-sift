PLAUSIBLE_OFFSET_MAX_S = 1000.0   # offsets beyond ±1000 s are implausible for NTP
PLAUSIBLE_NIST_DELTA_H = 24.0     # nist_time must be within 24 h of event_ts

class ValidationResult:
    def __init__(self, valid: bool, reason: str = ""):
        self.valid  = valid
        self.reason = reason
    def __bool__(self): return self.valid

def validate_context(ctx: NTPContext) -> ValidationResult:
    """Check plausibility of the NTPContext before enrichment."""
    if abs(ctx.ntp_offset_s) > PLAUSIBLE_OFFSET_MAX_S:
        return ValidationResult(False,
            f"ntp_offset_s={ctx.ntp_offset_s:.4f} exceeds ±{PLAUSIBLE_OFFSET_MAX_S} s")
    return ValidationResult(True)

def validate_enriched_row(row: dict) -> ValidationResult:
    """Post-enrichment row check: nist_time within 24 h of original event."""
    try:
        event_ts = _parse_event_ts(row)
        nist_ts  = datetime.strptime(row['nist_time'], '%Y-%m-%d %H:%M:%S.%fZ'
                   ).replace(tzinfo=timezone.utc)
        delta_h = abs((nist_ts - event_ts).total_seconds()) / 3600
        if delta_h > PLAUSIBLE_NIST_DELTA_H:
            return ValidationResult(False,
                f"nist_time delta {delta_h:.1f} h exceeds {PLAUSIBLE_NIST_DELTA_H} h")
    except Exception as exc:
        return ValidationResult(False, str(exc))
    return ValidationResult(True)

MAX_CORRECTION_ITERATIONS = 3

def validate_and_correct(csv_path: Path, ctx: NTPContext, output_path: Path,
                         iteration: int = 0) -> tuple[dict, NTPContext]:
    """
    Run enrich(). If validation fails and we are under the iteration cap,
    emit a correction log line and return (result, corrected_ctx) for the
    caller to re-run.

    Returns (result_dict, final_ctx).
    result_dict includes 'correction_needed': bool and 'correction_reason': str.
    """
    if iteration >= MAX_CORRECTION_ITERATIONS:
        # Cap reached — return a null result signalling escalation
        return ({
            'rows_processed': 0,
            'correction_needed': False,
            'correction_reason': 'max iterations reached — escalate to operator',
            'integrity_ok': False,
            'output_path': str(output_path),
        }, ctx)

    pre_check = validate_context(ctx)
    if not pre_check:
        print(f"[ntp-enrichment.self-correct] iter={iteration} "
              f"reason={pre_check.reason} action=reset-offset-to-zero",
              file=sys.stderr)
        corrected = NTPContext(ntp_source=ctx.ntp_source, ntp_offset_s=0.0,
                               ntp_assumption=True, confidence_rank=6)
        return ({'correction_needed': True,
                 'correction_reason': pre_check.reason,
                 'rows_processed': 0, 'integrity_ok': False,
                 'source_hash_before': '', 'source_hash_after': '',
                 'output_path': str(output_path)}, corrected)

    result = enrich(csv_path, ctx, output_path)
    result['correction_needed'] = False
    result['correction_reason'] = ''
    return result, ctx

test_validate_offset_rejects_implausible_large_value:
  ctx = NTPContext(ntp_source="dc01", ntp_offset_s=99999.0, ntp_assumption=True)
  vr = validate_context(ctx)
  assert not vr.valid
  assert "1000" in vr.reason

test_validate_nist_time_rejects_if_more_than_24h_from_event:
  row = {'date':'08/30/2018','time':'02:05:15','nist_time':'2019-01-01 02:05:15.000Z'}
  vr = validate_enriched_row(row)
  assert not vr.valid

test_self_correction_reruns_on_invalid:
  ctx_bad = NTPContext(ntp_source="dc01", ntp_offset_s=50000.0, ntp_assumption=True)
  result, corrected = validate_and_correct(mini_csv_path, ctx_bad,
                                            tmp_path/"out.csv", iteration=0)

  assert corrected.ntp_offset_s == 0.0
