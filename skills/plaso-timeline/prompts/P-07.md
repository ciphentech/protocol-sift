## P-07 — Manifest JSON + rubric evaluator

**Goal:** The manifest is the interface between the Python tools and the
agent's reasoning loop. Without it, the agent has no way to decide whether to
accept the result or self-correct. The rubric evaluator runs inside the Python
tool; the agent reads the output.

import json
from pathlib import Path

# Rubric thresholds (loaded from file if present; else these defaults)
_RUBRIC_DEFAULTS = {
    "ntp_offset_max_abs_s": 1.0,        # offsets > 1 s are suspicious for NTP
    "require_assumption_false": False,   # set True for legal submissions
    "require_eid_source": False,         # set True to require artifact-derived source
}

def _load_rubric(case_dir: Path) -> dict:
    rubric_path = case_dir / "analysis" / "ntp_rubric.json"
    if rubric_path.exists():
        return {**_RUBRIC_DEFAULTS, **json.loads(rubric_path.read_text())}
    return _RUBRIC_DEFAULTS.copy()

def _evaluate_rubric(summary: dict, rubric: dict) -> tuple[bool, list[str]]:
    """
    Evaluate the enrichment result against the rubric.
    Returns (rubric_pass: bool, rubric_failures: list[str]).
    """
    failures = []
    offset = abs(summary.get("representative_offset_s", 0.0))
    if offset > rubric["ntp_offset_max_abs_s"]:
        failures.append(
            f"ntp_offset_s abs value {offset:.4f} s exceeds rubric "
            f"threshold {rubric['ntp_offset_max_abs_s']} s — verify artifact"
        )
    if rubric["require_assumption_false"] and summary.get("ntp_assumption"):
        failures.append(
            "ntp_assumption=true but rubric requires confirmed source — "
            "provide --ntp-source or recover NTP logs"
        )
    if rubric["require_eid_source"] and summary.get("source_eid") is None:
        failures.append(
            "rubric requires EID-derived NTP source but none found in artifact"
        )
    return (len(failures) == 0), failures

def _suggested_action(failures: list[str]) -> str:
    if not failures:
        return ""
    if any("ntp_offset_s" in f for f in failures):
        return "recheck-offset"
    if any("ntp_assumption" in f or "EID" in f for f in failures):
        return "relax-assumption"
    return "escalate-to-operator"

def write_manifest(result: dict, ctx, output_path: Path,
                   iteration: int = 0) -> Path:
    """
    Write a manifest JSON alongside the enriched CSV.
    Path: ./analysis/<stem>_ntp_manifest[.iter<N>].json
    Returns the manifest path.
    """
    case_dir = output_path.parent.parent   # ../analysis/ relative to ../exports/
    case_dir.mkdir(parents=True, exist_ok=True)

    stem = output_path.stem
    suffix = f".iter{iteration}" if iteration > 0 else ""
    manifest_path = case_dir / "analysis" / f"{stem}_ntp_manifest{suffix}.json"
    manifest_path.parent.mkdir(parents=True, exist_ok=True)

    rubric = _load_rubric(case_dir)
    summary = {
        "rows_processed": result.get("rows_processed", 0),
        "ntp_source": ctx.ntp_source,
        "ntp_assumption": ctx.ntp_assumption,
        "confidence_rank": ctx.confidence_rank,
        "source_eid": ctx.source_eid,
        "representative_offset_s": ctx.ntp_offset_s,
        "integrity_ok": result.get("integrity_ok", False),
        "source_hash_before": result.get("source_hash_before", ""),
        "source_hash_after": result.get("source_hash_after", ""),
        "output_path": str(output_path),
        "iteration": iteration,
    }
    rubric_pass, rubric_failures = _evaluate_rubric(summary, rubric)
    summary["rubric_pass"] = rubric_pass
    summary["rubric_failures"] = rubric_failures
    summary["suggested_corrective_action"] = _suggested_action(rubric_failures)

    manifest_path.write_text(json.dumps(summary, indent=2))
    return manifest_path

def test_manifest_rubric_pass_on_valid_result():
    result = {'rows_processed': 14, 'integrity_ok': True,
               'source_hash_before': 'abc', 'source_hash_after': 'abc'}
    ctx = NTPContext(ntp_source="dc01", ntp_offset_s=-0.0004, ntp_assumption=False,
                     confidence_rank=1, source_eid=260)
    # Write manifest to tmp_path
    manifest = write_manifest(result, ctx, tmp_path/"exports"/"out.csv", iteration=0)
    data = json.loads(manifest.read_text())
    assert data["rubric_pass"] is True
    assert data["rubric_failures"] == []
    assert data["suggested_corrective_action"] == ""

def test_manifest_rubric_fails_on_large_offset():
    result = {'rows_processed': 14, 'integrity_ok': True,
               'source_hash_before': 'abc', 'source_hash_after': 'abc'}
    ctx = NTPContext(ntp_source="dc01", ntp_offset_s=500.0, ntp_assumption=False,
                     confidence_rank=1, source_eid=260)
    manifest = write_manifest(result, ctx, tmp_path/"exports"/"out.csv")
    data = json.loads(manifest.read_text())
    assert data["rubric_pass"] is False
    assert len(data["rubric_failures"]) > 0
    assert data["suggested_corrective_action"] == "recheck-offset"

def test_manifest_iteration_suffix():
    result = {'rows_processed': 14, 'integrity_ok': True,
               'source_hash_before': 'abc', 'source_hash_after': 'abc'}
    ctx = NTPContext(ntp_source="dc01")
    m1 = write_manifest(result, ctx, tmp_path/"exports"/"out.csv", iteration=1)
    m2 = write_manifest(result, ctx, tmp_path/"exports"/"out.csv", iteration=2)
    assert ".iter1." in str(m1)
    assert ".iter2." in str(m2)
    assert m1 != m2   # never overwritten

def test_manifest_schema_has_required_keys():
    result = {'rows_processed': 5, 'integrity_ok': True,
               'source_hash_before': 'x', 'source_hash_after': 'x'}
    ctx = NTPContext()
    manifest = write_manifest(result, ctx, tmp_path/"exports"/"out.csv")
    data = json.loads(manifest.read_text())
    required = {"rows_processed","ntp_source","ntp_assumption","confidence_rank",
                "integrity_ok","rubric_pass","rubric_failures",
                "suggested_corrective_action","iteration"}
    assert required.issubset(data.keys())
